def update(dir)
  case dir
  when 'l'
    @box.top -= 2
    @box.left -= 2
    @box.bottom -= 2
    @box.right -= 2
  when 'r'
    @box.top += 2
    @box.left += 2
    @box.bottom += 2
    @box.right += 2
  when 'u'
    @z += 1
  when 'd'
    @z -= 1
  end
end
